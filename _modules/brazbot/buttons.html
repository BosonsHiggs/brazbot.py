[docs]    def to_dict(self):
        data = {
            "type": 2,
            "label": self.label,
            "style": self.style,
            "custom_id": self.custom_id,
            "disabled": self.disabled
        }

        if self.url:
            data["url"] = self.url
        if self.emoji:
            data["emoji"] = self.emoji

        return data


    @property
    def to_component(self):
        return {"type": 1, "components": [self.to_dict()]}
